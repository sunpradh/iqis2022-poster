\block[bodyverticalshift=5mm]{}{\Large
We study the Quantum Fisher Information (QFI) in one-dimensional models as a tool for measuring the Multipartite Entanglement (ME), which can give valuable information about the existence of topological phases.
We show that the scaling of the QFI of strictly non-local observables can be used for characterizing the phase diagrams and, in particular, for detecting topological phases, where it scales maximally.
}

            \end{equation*}

    %----------------------------------------------------------------------------------------------------
    \myblock{QUANTUM FISHER INFORMATION}{
        % Unitary transformation with \textbf{unknown angle} $\theta$
        \Highlight{PHASE ESTIMATION}
        \begin{equation*}
            \rho \to \rho(\theta) = e^{-i \theta \hat{H}} \rho e^{i \theta \hat{H}}
        \end{equation*}

        Angle $\theta$ to be estimated with $m$ measurements and operators $\{\hat{E}_{\mu}\}$

        \vspace*{0.5cm}

        \innerblock{Quantum Cramér-Rao bound}{
            \begin{equation*}
                (\Delta \theta)^2 \geq
                \frac{1}{m F[\rho(\theta), \{\hat{E}_{\mu}\}]} \geq
                \frac{1}{m F_Q[\rho, \hat{H}]}
            \end{equation*}

            \InnerNote{limits the precision of the estimation of $\theta$}
        }

        \vspace*{1cm}

        \innerblock{Fisher information}{
            \begin{equation*}
                F[\rho(\theta), \{\hat{E}_{\mu}\}] = \sum_{\mu} \frac{ [\partial_{\theta} P(\mu|\theta)]^2}{P(\mu | \theta)}
                \qquad
                P(\mu | \theta) = \Tr[\rho(\theta) \hat{E}_{\mu}]
            \end{equation*}

            \InnerNote{$P(\mu | \theta)$ conditional probabilities of a measure $\mu$ given $\theta$}
        }

        \vspace*{1cm}

        \innerblock{Quantum Fisher information}{
            \begin{equation*}
                F_Q[\rho, \hat{H}] = \max_{\{\hat{E}_{\mu}\}} F[\rho(\theta), \{\hat{E}_{\mu}\}]
            \end{equation*}

            \InnerNote{maximum of $F$ over the set of possible measures $\{\hat{E}_{\mu}\}$}
        }

        \vspace*{1cm}
        For pure states:~~~~$F_Q[\ket{\psi}, \hat{H}] = 4 (\Delta \hat{H})^2 = 4(\ev*{\hat{H}^2}_{\psi} - \ev*{\hat{H}}_{\psi}^2)$
    }

    %----------------------------------------------------------------------------------------------------
    \myblock{ENTANGLEMENT CRITERION}{

        \Highlight{Input state with k-party entanglement}~~~$\rho_{\text{k-ent}}$

        \vspace*{0.5cm}
        Linear two-mode interferometer:
        \vspace*{0.5cm}
        \begin{equation*}
            \hat{H}_{\text{lin}}
            = \frac{1}{2} \sum_{j} \vec{n}_j \cdot \vec{\sigma}_j
            = \frac{1}{2} \sum_{j} \qty( \alpha_j \sigma_j^ x + \beta_j \sigma_j^y + \gamma_j \sigma_j^z )
        \end{equation*}

        \vspace*{0.5cm}


        \innerblock{$F_Q$ Criterion}{
            \begin{equation*}
                F_Q[\rho_{\text{k-ent}}, \hat{H}_{\text{lin}}] \leq k N
            \end{equation*}
        }
        \begin{center}
            \begin{tikzpicture}
                \node (sep) at (0,0) [nota] {$F_Q[\rho_{\text{sep}}, \hat{H}] \leq N$};
                \node (max) [nota, right=8cm of sep] {$F_Q[\rho_{\text{max}}, \hat{H}] \leq N^2$};
                \TitoloNota{sep}{fully separable}
                \TitoloNota{max}{maximally entangled}
            \end{tikzpicture}
        \end{center}
    }
